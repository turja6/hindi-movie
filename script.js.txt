const movies = [
    { title: "Inception", description: "A mind-bending thriller by Christopher Nolan." },
    { title: "Interstellar", description: "A sci-fi adventure through space and time." },
    { title: "The Dark Knight", description: "Batman faces the Joker in Gotham City." }
];

const moviesContainer = document.getElementById("movies-container");
const modal = document.getElementById("modal");
const modalTitle = document.getElementById("movie-title");
const modalDescription = document.getElementById("movie-description");
const closeModal = document.querySelector(".close");

movies.forEach(movie => {
    const movieElement = document.createElement("div");
    movieElement.classList.add("movie");
    movieElement.innerHTML = `<h3>${movie.title}</h3>`;
    movieElement.addEventListener("click", () => {
        modal.style.display = "flex";
        modalTitle.textContent = movie.title;
        modalDescription.textContent = movie.description;
    });
    moviesContainer.appendChild(movieElement);
});

closeModal.addEventListener("click", () => {
    modal.style.display = "none";
});

window.addEventListener("click", (e) => {
    if (e.target === modal) {
        modal.style.display = "none";
    }
});
